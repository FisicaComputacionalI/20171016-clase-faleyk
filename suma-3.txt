#include <iostream>

using namespace std;

int main()
{
    int sum=0, count =1;

    while (count < 10) {
        if ((count%2)>0){
            sum=sum+count;
        }
        count++;
    }
    cout << "Sum = " << sum<<endl;
    return 0;

}
